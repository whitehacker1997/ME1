//Local
		Scaffold-DBContext "Host=localhost;Username=postgres;Password=postgres;Database=e_center" Npgsql.EntityFrameworkCore.PostgreSQL -DataAnnotations -OutputDir AutoGenerated -Namespace OnApp.DataLayer.PgSql -StartupProject OnApp.DataLayer.PgSql -Project OnApp.DataLayer.PgSql

//Real
		Scaffold-DBContext "Host=my-database-instance.c3cm2cki4uqk.ap-southeast-1.rds.amazonaws.com;Username=postgres;Password=postgres;Database=e_center;Pooling=true" Npgsql.EntityFrameworkCore.PostgreSQL -DataAnnotations -OutputDir AutoGenerated -Namespace OnApp.DataLayer.PgSql -StartupProject OnApp.DataLayer.PgSql -Project OnApp.DataLayer.PgSql

3) Select all files in "AutoGenerated" folder and set theirs Build Action -> None
